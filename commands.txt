# 🏠 Commandes Utiles - Jumeau Numérique Smart Home

## 🚀 Démarrage du Système
# Démarrer les serveurs (API, MQTT, DB)
sudo systemctl start docker
export DOCKER_HOST=unix:///var/run/docker.sock
docker compose up -d

# Lancer la simulation automatique (Capteurs aléatoires)
source backend/venv/bin/activate
python3 test_mqtt_sim.py &

## 🌡️ Simulation Manuelle (Contrôle par zone)
# Simuler une vague de chaleur (Chaud dans le salon)
python3 -c "import paho.mqtt.client as mqtt, json; c=mqtt.Client(mqtt.CallbackAPIVersion.VERSION2); c.connect('localhost'); c.publish('home/sensors', json.dumps({'salon': {'temperature': 32.0, 'presence': True, 'luminosity': 800}, 'cuisine': {'temperature': 24.0, 'presence': False, 'luminosity': 200}})); c.disconnect()"

# Simuler un froid intense (Partout)
python3 -c "import paho.mqtt.client as mqtt, json; c=mqtt.Client(mqtt.CallbackAPIVersion.VERSION2); c.connect('localhost'); c.publish('home/sensors', json.dumps({'salon': {'temperature': 15.0, 'presence': True, 'luminosity': 400}, 'cuisine': {'temperature': 14.0, 'presence': True, 'luminosity': 150}})); c.disconnect()"

## 💡 Simulation de Luminosité & Présence
# Simuler le matin (Lumière dans la cuisine, personne au salon)
python3 -c "import paho.mqtt.client as mqtt, json; c=mqtt.Client(mqtt.CallbackAPIVersion.VERSION2); c.connect('localhost'); c.publish('home/sensors', json.dumps({'salon': {'temperature': 21.0, 'presence': False, 'luminosity': 100}, 'cuisine': {'temperature': 22.0, 'presence': True, 'luminosity': 1200}})); c.disconnect()"

# Simuler la soirée (Présence au salon, faible luminosité)
python3 -c "import paho.mqtt.client as mqtt, json; c=mqtt.Client(mqtt.CallbackAPIVersion.VERSION2); c.connect('localhost'); c.publish('home/sensors', json.dumps({'salon': {'temperature': 22.0, 'presence': True, 'luminosity': 50}, 'cuisine': {'temperature': 20.0, 'presence': False, 'luminosity': 10}})); c.disconnect()"

## 🌡️ Contrôle Thermostat (Température de Régulation)
# Régler la consigne du salon à 25°C
curl -X POST "http://localhost:8000/control/salon/climatisation/target_temp?temp=25"

# Régler la consigne de la cuisine à 20°C
curl -X POST "http://localhost:8000/control/cuisine/climatisation/target_temp?temp=20"

# Forcer le mode MANUAL (COOL) dans le salon
curl -X POST "http://localhost:8000/control/salon/climatisation/set?value=COOL"

# Repasser le salon en AUTO
curl -X POST "http://localhost:8000/control/salon/climatisation/mode?mode=AUTO"

## 💡 Contrôle Éclairage
# Allumer la lumière du salon
curl -X POST "http://localhost:8000/control/salon/light?state=true"

# Éteindre la lumière de la cuisine
curl -X POST "http://localhost:8000/control/cuisine/light?state=false"

## ⚡ Gestion Énergétique (Basculement Solaire/Réseau)
# Simuler une batterie CRITIQUE (Passage sur RÉSEAU)
curl -X POST "http://localhost:8000/control/energy/battery?level=15"

# Simuler une batterie PLEINE (Repassage sur SOLAIRE)
curl -X POST "http://localhost:8000/control/energy/battery?level=100"

## ⏹️ Arrêt du Système
# Arrêter le simulateur
pkill -f test_mqtt_sim.py

# Arrêter les serveurs Docker
docker compose down
